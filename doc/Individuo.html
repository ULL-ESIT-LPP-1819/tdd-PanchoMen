def media(array)
        var = 0
        count = 0
        while var < array.length
                count += array[var]
                var += 1
        end
        return (count/array.length)
end

def to_s
		string = "\t#{@nombre}\n"
		string += "peso: #{@peso}\n"
		string += "talla: #{@talla}\n"
		string += "edad: #{@edad}\n"
		string += "sexo: #{get_sexo}\n"
		string += "media circunferencia de cintura: #{@cir_cintura}\n"
		string += "media circunferencia de cadera: #{@cir_cadera}"
		return string
end
